async function sendMsg() { const input = document.getElementById("userInput"); const msg = input.value; if(!msg) return; const chatBox = document.getElementById("chatBox"); chatBox.innerHTML += `
${msg}
`; input.value = ""; try { // URL ko dhyan se dekhein: hum relative path use kar rahe hain const res = await fetch("/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ msg: msg }) // Sirf msg bhej rahe hain }); const data = await res.json(); const reply = data.reply || "Error: " + (data.error || "Unknown"); chatBox.innerHTML += `
${reply}
`; chatBox.scrollTop = chatBox.scrollHeight; } catch (e) { chatBox.innerHTML += `
Connection failed!
`; } }
